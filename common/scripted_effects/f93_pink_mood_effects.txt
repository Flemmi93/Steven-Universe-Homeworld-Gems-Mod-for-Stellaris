increase_pink_mood_10 = {
	increase_pink_mood = { ADD = 10 }
}
#increases the mood of pink by the number defined as ADD, example above would be 10
increase_pink_mood = {
	root.owner = {
		if = {
			limit = {
				has_event_chain = pink_event_chain
				is_variable_set = f93_pinks_mood_variable
			}
			root.owner = {
				add_event_chain_counter = {
					event_chain = pink_event_chain
					counter = f93_pinks_mood_counter
					amount = $ADD$
				}
				change_variable = { which = f93_pinks_mood_variable value = $ADD$ }
			}
		}
	}
}

export_X_month_resource_income_to_variable = {
	root.owner = {
		export_resource_income_to_variable = {
			resource = $resource$
			variable = $variable$
		}
		multiply_variable = { which = $variable$ value = $month$ }
	}
}
