f93_pinks_garden = {
	entity = "habitat_phase_01_entity"
	construction_entity = "habitat_phase_01_entity"
	portrait = "GFX_megastructure_habitat_background"
	place_entity_on_planet_plane = yes
	show_galactic_map_icon = no
	hide_name = no
	show_in_outliner = no
	build_time = 1080

	resources = {
		category = megastructures_habitat
		cost = {
			influence = 150
		}
		cost = {
			energy = 2000
			consumer_goods = 500
			alloys = 1000
		}
	}
	construction_blocks_and_blocked_by = self_type
	potential = {
		has_country_flag = f93_pink_garden_unlock
	}

	possible = {
		hidden_trigger = {
			exists = starbase
		}
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
	}
	
	#Avoid the possability of 2 being build.
	on_build_start = { owner = { remove_country_flag = f93_pink_garden_unlock } }
	on_build_cancel = { owner = { set_country_flag = f93_pink_garden_unlock } }

	on_build_complete = {
		owner = { remove_country_flag = f93_pink_garden_unlock }
		remove_megastructure = fromfrom
		reroll_random = yes #in case it is badly placed reload right before the construction completes
		random_list = {
			1 = {
				spawn_megastructure = {
					type = f93_pinks_garden2
					graphical_culture = from
					planet = star
					orbit_distance = 75
					orbit_angle = 45
				}
			}
			1 = {
				spawn_megastructure = {
					type = f93_pinks_garden2
					graphical_culture = from
					planet = star
					orbit_distance = 80
					orbit_angle = 295
				}
			}
			1 = {
				spawn_megastructure = {
					type = f93_pinks_garden2
					graphical_culture = from
					planet = star
					orbit_distance = 85
					orbit_angle = 165
				}
			}
		}
		from = {
			random_situation = {
				limit = { is_situation_type = pink_situation }
				add_situation_progress = 25
			}
		}
	}
}
