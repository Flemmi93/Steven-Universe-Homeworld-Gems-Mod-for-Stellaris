	id = gem_diamond_ascension.0001
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		random_planet_within_border = {
			limit = { NOT = { exists = archaeological_site } }
			create_archaeological_site = white_ascension_dig_site
		}
		random_planet_within_border = {
			limit = { NOT = { exists = archaeological_site } }
			create_archaeological_site = yellow_ascension_dig_site
		}
		random_planet_within_border = {
			limit = { NOT = { exists = archaeological_site } }
			create_archaeological_site = blue_ascension_dig_site
		}
		random_planet_within_border = {
			limit = { NOT = { exists = archaeological_site } }
			create_archaeological_site = pink_ascension_dig_site
		}
	}
}
