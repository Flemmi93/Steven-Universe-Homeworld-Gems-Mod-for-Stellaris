#event to choose Gem Empire Midgame Crisis
country_event = {
	id = GEM_start.1005
	title = "GEM_start.1005.name"
	desc = "GEM_start.1005.desc"
	picture = GFX_evt_gem_homeworld
	show_sound = event_mystic_reveal
	is_triggered_only = yes
	fire_only_once = yes

	trigger = {
		is_geo_empire = no
		is_ai = no
	}
	immediate = {
		get_galaxy_setup_value = {
			setting = mid_game_year
			which = partway_to_mid_game_years_var
			scale_by = 0.75
		}
	}
	option = {
		name = YES
		allow = { NOT = { has_global_flag = choose_gem_invasion } }
		hidden_effect = {
			set_country_flag = choose_gem_invasion_cf
			random_system = {
				limit = {
					has_owner = no
					is_rim_system = yes
					NOT = {
						any_neighbor_system = {
							has_owner = yes
						}
					}
				}
				save_global_event_target_as = Gem_midgame_crisis_target
				set_global_flag = choose_gem_invasion
			}
		}
	}
	option = {
		name = NO
	}
	option = {
		name = GEM_start.1005.b
		hidden_effect = {
			if = {
				limit = {
					any_system = {
						any_system_planet = {
							has_planet_flag = planet_earth
							OR = {
								is_colony = no
								owner = {
									is_primitive = yes
								}
							}
							owner = { is_ai = yes }
						}
					}
				}
				random_system = {
					limit = { any_system_planet = { has_planet_flag = planet_earth } }
					random_system_planet = {
						limit = { is_moon = yes orbit = { has_planet_flag = planet_earth } }
						save_global_event_target_as = gem_spawn_location
						change_pc = pc_gem
						create_gem_empire_normal = yes
						set_name = "Luna"
					}
				}
			}
			else_if = {
				limit = {
					any_system = {
						any_system_planet = {
							has_planet_flag = planet_earth
							owner = {
								has_origin = origin_default
								is_ai = yes
							}
						}
					}
				}
				random_system = {
					limit = { any_system_planet = { has_planet_flag = planet_earth } owner = { has_origin = origin_default is_ai = yes } }
					random_system_planet = {
						limit = { is_moon = yes orbit = { has_planet_flag = planet_earth } }
						save_global_event_target_as = gem_spawn_location
						change_pc = pc_gem
						create_gem_empire_normal = yes
						set_name = "Luna"
						orbit = {
							# set_variable = { which = fraction_num_pops value = trigger:count_owned_pop_amount }
							# change_variable = { which = fraction_num_pops value = -10 }
							# while = {
							# 	count = fraction_num_pops
							# 	kill_pop_group = {
							# 		pop_group = this
							# 		amount = 1
							# 	}
							# }
							
							set_owner = last_created_country
							save_event_target_as = planet_earth
							add_modifier = { modifier = pm_f93_human_preserve }
						}
						while = {
							count = 10
							kill_pop_group = {
								pop_group = this
								amount = 1
							}
						}
						while = {
							count = 3
							random_owned_pop_group = { resettle_pop = { pop = this planet = event_target:planet_earth } }
						}

					}
				}
			}
			else_if = {
				random_country = {
					limit = {
						is_ai = yes
						has_origin = origin_default
						is_country_type = default
						has_federation = no
					}
					home_planet = {
						save_global_event_target_as = gem_spawn_location
						change_pc = pc_gem
						owner = { destroy_country = yes }
						while = {
							count = 60
							kill_pop_group = {
								pop_group = this
								amount = 1
							}
						}
						create_gem_empire_normal = yes
						set_name = Homeworld
						solar_system = { set_name = Luster }
					}
				}
			}
			else = {
				random_system = {
					set_name = Luster
					star = { set_name = Luster }
					limit = {
						NOT = { any_fleet_in_system = { exists = this } }
						any_system_planet = {
							is_colonizable = yes
							NOT = { is_planet_class = pc_gaia }
						}
						any_neighbor_system = {
							NOT = { any_fleet_in_system = { exists = this } }
							any_neighbor_system = {
								NOT = { any_fleet_in_system = { exists = this } }
							}
						}
					}
					random_system_planet = {
						limit = { is_colonizable = yes }
						save_global_event_target_as = gem_spawn_location
						change_pc = pc_gem
						create_gem_empire_normal = yes
						set_name = Homeworld
					}
				}
			}
		}
	}
	option = {
		name = "gem_start.1005.c"
		custom_tooltip = "gem_start.1005.c.desc" 
		allow = { NOT = { has_global_flag = choose_gem_invasion } }
		hidden_effect = {
			random_list = {
				50 = {
					set_country_flag = choose_gem_invasion_cf
					random_system = {
						limit = {
							has_owner = no
							is_rim_system = yes
							NOT = {
								any_neighbor_system = {
									has_owner = yes
									NOT = {
										any_neighbor_system = {
											has_owner = yes
										}
									}
								}
							}
						}
						save_global_event_target_as = Gem_midgame_crisis_target
						set_global_flag = choose_gem_invasion
					}
				}
				50 = {
				}
			}
		}
	}
