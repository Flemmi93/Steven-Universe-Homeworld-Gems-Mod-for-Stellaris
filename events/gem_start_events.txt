			portrait="gem"

			plural = "Gems"
			name_list="GEO1"
			gender=not_set
			traits = {
				trait="trait_geoloid"
				trait="trait_crystal_gems"
				trait="trait_gem_holobody"
				trait="trait_gem_hardness"
				trait="trait_gem_donteat"
			}

		}

			save_event_target_as = crystal_gems_species

		}
	}
