namespace = gem_kg


#To inform about the kindergarten mechanic
planet_event = {
	id = gem_kg.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_geo_empire = yes
	}

	immediate = {
		owner = { country_event = { id = gem_kg.2 days = 1 } }
	}
}

country_event = {
	id = gem_kg.2
	title = "gem_kg.2.name"
	desc = gem_kg.2.desc
	picture = "GFX_evt_gem_kindergarten"
	show_sound = event_mystic_reveal
	is_triggered_only = yes

	#every time a colony is founded and they havent converted Farms to Kindergarten deposits they should get this
	trigger = {
		is_geo_empire = yes
		owner = {
			NOT = { has_country_flag = gem_dontshow_kg_tut } 
		}
	}

	option = { 
	name = OK 
	}
	option = {
	name = "gem_kg.2.b"
			if = {
			limit = {
				NOT = { has_country_flag = gem_dontshow_kg_tut }
			}
			set_country_flag = gem_dontshow_kg_tut
		}
	}
}

planet_event = {
	id = gem_kg.5
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		if = {
			limit = { From = { is_geo_empire = no } }
			every_deposit = {
				limit = { is_deposit_type = d_kindergarten }
				root = { add_deposit = d_kindergarten_blocker }
				remove_deposit = yes
			}
		}
		else_if = {
			limit = { From = { is_geo_empire = yes } }
			every_deposit = {
				limit = { is_deposit_type = d_kindergarten_blocker }
				root = { add_deposit = d_kindergarten }
				remove_deposit = yes
			}
		}
	}
}
